CREATE TABLE LOGIN (userId VARCHAR(20), password VARCHAR(12), authorityLevel INTEGER, CRDATE TIMESTAMP, UPDDATE TIMESTAMP, PRIMARY KEY (userId))

CREATE TABLE PRODUCT (author VARCHAR(50), imageURL VARCHAR(255), imageLargeURL VARCHAR(255), manufacturer VARCHAR(50), productgroup VARCHAR(20), productId VARCHAR(20), productidtype VARCHAR(20), source VARCHAR(20), sourceid VARCHAR(20), title VARCHAR(100), crDate TIMESTAMP, updDate TIMESTAMP, PRIMARY KEY ( productId )) 

CREATE TABLE REGISTEREDUSER (id INT NOT NULL AUTO_INCREMENT, userId  VARCHAR(20), email VARCHAR(50), firstname VARCHAR(50), lastname VARCHAR(50), telephone VARCHAR(50), CRDATE TIMESTAMP, UPDDATE TIMESTAMP, PRIMARY KEY ( id ))

CREATE TABLE USERTOPRODUCT  (id INT NOT NULL AUTO_INCREMENT, userId VARCHAR(20), productId VARCHAR(20), productIndex INT, CRDATE TIMESTAMP,  UPDDATE TIMESTAMP,  PRIMARY KEY ( id ))
